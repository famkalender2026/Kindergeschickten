import React, { useState } from "react";
import {
  ActivityIndicator,
  ScrollView,
  StyleSheet,
  Text,
  TextInput,
  TouchableOpacity,
  View,
} from "react-native";

export default function HomeScreen(): React.JSX.Element {
  const [alter, setAlter] = useState<string>("5");
  const [akteure, setAkteure] = useState<string>("");
  const [genre, setGenre] = useState<string>("Abenteuer");
  const [geschichte, setGeschichte] = useState<string>("");
  const [laedt, setLaedt] = useState<boolean>(false);

  const genres: string[] = [
    "Abenteuer",
    "Märchen",
    "Tiere",
    "Weltraum",
    "Detektiv",
  ];

  const generiereGeschichte = async (): Promise<void> => {
    setLaedt(true);
    setGeschichte("");
    const prompt: string = `Schreibe eine Kindergeschichte auf Deutsch. Alter: ${alter} Jahre. Akteure: ${akteure || "ein kleines Kaninchen"}. Genre: ${genre}. Die Geschichte soll spannend, lehrreich und kindgerecht sein. Beginne direkt mit der Geschichte.`;
    try {
      const antwort = await fetch(
        "https://api.groq.com/openai/v1/chat/completions",
        {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            Authorization:
              "Bearer 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",
          },
          body: JSON.stringify({
            model: "llama-3.3-70b-versatile",
            messages: [{ role: "user", content: prompt }],
            max_tokens: 1000,
          }),
        },
      );
      const daten = await antwort.json();
      setGeschichte(JSON.stringify(daten));
    } catch (fehler) {
      setGeschichte("Ups! Da ist etwas schiefgelaufen. Versuche es nochmal.");
    }
    setLaedt(false);
  };

  return (
    <ScrollView style={styles.container}>
      <Text style={styles.titel}>Geschichten-Zauber</Text>
      <Text style={styles.label}>Alter des Kindes</Text>
      <TextInput
        style={styles.input}
        value={alter}
        onChangeText={setAlter}
        keyboardType="numeric"
        placeholder="z.B. 5"
      />
      <Text style={styles.label}>Akteure</Text>
      <TextInput
        style={styles.input}
        value={akteure}
        onChangeText={setAkteure}
        placeholder="z.B. ein tapferer Ritter und ein Drache"
      />
      <Text style={styles.label}>Genre</Text>
      <View style={styles.chipReihe}>
        {genres.map((g: string) => (
          <TouchableOpacity
            key={g}
            style={[styles.chip, genre === g && styles.chipAktiv]}
            onPress={() => setGenre(g)}
          >
            <Text style={genre === g ? styles.chipTextAktiv : styles.chipText}>
              {g}
            </Text>
          </TouchableOpacity>
        ))}
      </View>
      <TouchableOpacity style={styles.button} onPress={generiereGeschichte}>
        {laedt ? (
          <ActivityIndicator color="white" />
        ) : (
          <Text style={styles.buttonText}>Geschichte erfinden!</Text>
        )}
      </TouchableOpacity>
      {geschichte ? (
        <View style={styles.geschichteBox}>
          <Text style={styles.geschichteText}>{geschichte}</Text>
        </View>
      ) : null}
    </ScrollView>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, padding: 20, backgroundColor: "#FFF8F0" },
  titel: {
    fontSize: 28,
    fontWeight: "bold",
    textAlign: "center",
    marginVertical: 20,
    color: "#8B4513",
  },
  label: {
    fontSize: 16,
    fontWeight: "600",
    marginTop: 16,
    marginBottom: 6,
    color: "#555",
  },
  input: {
    borderWidth: 1,
    borderColor: "#DDD",
    borderRadius: 10,
    padding: 12,
    fontSize: 16,
    backgroundColor: "white",
  },
  chipReihe: {
    flexDirection: "row",
    flexWrap: "wrap",
    gap: 8,
    marginBottom: 8,
  },
  chip: {
    borderWidth: 1,
    borderColor: "#DDD",
    borderRadius: 20,
    paddingHorizontal: 16,
    paddingVertical: 8,
  },
  chipAktiv: { backgroundColor: "#FF8C42", borderColor: "#FF8C42" },
  chipText: { color: "#666" },
  chipTextAktiv: { color: "white", fontWeight: "600" },
  button: {
    backgroundColor: "#FF8C42",
    borderRadius: 14,
    padding: 16,
    marginTop: 24,
    alignItems: "center",
  },
  buttonText: { color: "white", fontSize: 18, fontWeight: "bold" },
  geschichteBox: {
    marginTop: 24,
    backgroundColor: "white",
    borderRadius: 14,
    padding: 20,
    marginBottom: 40,
  },
  geschichteText: { fontSize: 16, lineHeight: 26, color: "#333" },
});
